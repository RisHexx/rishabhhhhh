FROM python:3.10-slim
WORKDIR /app
COPY . .
CMD ["python", "main.py"]

public class HelloWorld{
    public static void main(String[] args) {
        System.out.println("Hello, World!");
    }
}

pipeline {
    agent any

    stages {
        stage('build') {
            steps {
                bat 'javac HelloWorld.java'
            }
        }
        stage('run') {
            steps {
                bat 'java HelloWorld'
            }
        }
    }
}
